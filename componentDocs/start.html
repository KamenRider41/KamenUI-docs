# 快速开始

# 1.安装组件库

npm i kamen-ui

# 2.引用组件库

在main.js中引用组件库

// 全部引入
import { createApp } from 'vue'
import 'kamen-ui/dist/css/index.css'
import KUI from 'kamen-ui'
let Vue = createApp(App)
Vue.use(KUI)

// 按需引用(有待优化)
import { createApp } from 'vue'
import 'kamen-ui/dist/css/button.css'
import KUI from 'kamen-ui'
let Vue = createApp(App)
Vue.use(KUI.Button)
